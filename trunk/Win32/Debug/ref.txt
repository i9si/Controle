	SELECT    pvi.produto_id,    p.descricao,		pvi.quantidade,		pvi.valor_unitario,		pvi.subtotal,		pvi.desconto,		pvi.valor_total,		to_char(pvi.data_inclusao,'dd/mm/yyyy') data_inclusao,		pvi.numero_item_cupom,		PVI.CANCELADO,    PVI.PEDIDO_VENDA_ID,    PES.NOME_FANTASIA NOME_VENDEDOR	FROM GESTOR.pedido_venda_item pvi  INNER JOIN GESTOR.produto p ON pvi.produto_id = p.id  INNER JOIN GESTOR.pedido_venda pv ON PVI.PEDIDO_VENDA_ID = PV.ID  LEFT JOIN GESTOR.VENDEDOR VEN ON VEN.ID = PV.VENDEDOR_ID  LEFT JOIN GESTOR.PESSOA PES ON PES.ID = VEN.ID	where PVI.PEDIDO_VENDA_ID = :id
